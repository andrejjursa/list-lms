  310.      * @param Student|Teacher$recipients list of students or teachers.

  318.     protected function _send_multiple_emails($recipients, $subject, $template, $template_variables = array(), $from = NULL, $from_name = '') {

  333.             $result = TRUE;
  334.             $lang_clone = clone $this->lang;

  348.                     $partial_result = $this->email->send();
  349.                     $result = $result && $partial_result;
