  56.         $task_set = new Task_set();

  62.             if (is_array($data) && count($data) > 0) { foreach ($data as $student_id => $solution_data) {

  66.                 $task_set_check->where_related('course/participant/student', 'id', intval($student_id));

  68.                 $task_set_check->group_start();
  69.                     $task_set_check->or_where('group_id', NULL);

  74.                     $solution = new Solution();
  75.                     $solution->where_related_student('id', $student->id);
  76.                     $solution->where_related_task_set('id', $task_set->id);

  79.                         $solution->teacher_id = $this->usermanager->get_teacher_id();

  162.                         } else {
  163.                             $this->db->trans_rollback();
  164.                             $result->message = $this->lang->line('admin_solutions_remove_points_error_task_set_upload_limit_not_reached');

  706.             foreach ($students as $student) {
